// Initialize the editor
var editor = new JSONEditor(document.getElementById("editor_holder"),{
   schema: {...},
   theme: 'bootstrap2'
});

// Set the value
editor.setValue({...});

// Get the value
var data = editor.getValue();

// Validate
var errors = editor.validate();
if(errors.length) {
  // Not valid
}

// Listen for changes
editor.on("change",  function() {
  // Do something...
});
